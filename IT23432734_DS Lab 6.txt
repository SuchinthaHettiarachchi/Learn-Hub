-- PART 2: XPATH EXPRESSIONS

-- XPath Expression 1: Get all product elements
-- /catalog/product - Absolute path from root
SELECT id, xDoc.query('/catalog/product') AS products
FROM AdminDocs;

-- XPath Expression 2: Get all product elements (anywhere in document)
-- //product - Descendant-or-self axis
SELECT id, xDoc.query('//product') AS products
FROM AdminDocs;

-- XPath Expression 3: Get products using wildcard for parent
-- /*/product - Any child of root, then product
SELECT id, xDoc.query('/*/product') AS products
FROM AdminDocs;

-- XPath Expression 4: Get products in WMN department using attribute filter
-- [@dept="WMN"] - Attribute predicate
SELECT id, xDoc.query('/*/product[@dept="WMN"]') AS wmn_products
FROM AdminDocs;

-- XPath Expression 5: Get WMN products using explicit axis syntax
-- child::product[attribute::dept="WMN"]
SELECT id, xDoc.query('/*/child::product[attribute::dept="WMN"]') AS wmn_products
FROM AdminDocs;

-- XPath Expression 6: Get WMN products using descendant axis (INCORRECT - should filter attributes)
-- Note: This may not work as expected because dept is an attribute, not an element
SELECT id, xDoc.query('//product[dept="WMN"]') AS wmn_products
FROM AdminDocs;

-- XPath Expression 7: Get WMN products using descendant-or-self axis
SELECT id, xDoc.query('descendant-or-self::product[attribute::dept="WMN"]') AS wmn_products
FROM AdminDocs;

-- XPath Expression 8: Get products where number > 500
SELECT id, xDoc.query('//product[number > 500]') AS products_gt_500
FROM AdminDocs
WHERE id = 1;

-- XPath Expression 9: Get product numbers > 500
-- [. gt 500] - Current node value greater than 500
SELECT id, xDoc.query('//product/number[. gt 500]') AS numbers_gt_500
FROM AdminDocs
WHERE id = 1;

-- XPath Expression 10: Get the 4th product
-- [4] - Position predicate
SELECT id, xDoc.query('/catalog/product[4]') AS fourth_product
FROM AdminDocs
WHERE id = 1;

-- XPath Expression 11: Get ACC products with number > 500
-- Multiple predicates
SELECT id, xDoc.query('//product[number > 500][@dept="ACC"]') AS acc_products_gt_500
FROM AdminDocs
WHERE id = 1;

-- XPath Expression 12: Get first product with number > 500
-- [1] after filter
SELECT id, xDoc.query('//product[number > 500][1]') AS first_product_gt_500
FROM AdminDocs
WHERE id = 1;


-- PART 3: XQUERY EXPRESSIONS (FLWOR)

-- FLWOR = For, Let, Where, Order by, Return

-- XQuery 1: Get all product numbers
-- Basic FLWOR expression
SELECT xDoc.query(' 
    for $prod in //product 
    let $x := $prod/number
    return $x
') AS product_numbers
FROM AdminDocs
WHERE id = 1;

-- XQuery 2: Get product numbers > 500 with WHERE clause
SELECT xDoc.query(' 
    for $prod in //product 
    let $x := $prod/number
    where $x > 500
    return $x
') AS numbers_gt_500
FROM AdminDocs
WHERE id = 1;

-- XQuery 3: Same as XQuery 1 (duplicate)
SELECT xDoc.query(' 
    for $prod in //product 
    let $x := $prod/number
    return $x
') AS product_numbers
FROM AdminDocs
WHERE id = 1;

-- XQuery 4: Get numbers > 500 wrapped in <Item> element
SELECT xDoc.query(' 
    for $prod in //product 
    let $x := $prod/number
    where $x > 500
    return (<Item>{$x}</Item>)
') AS items_gt_500
FROM AdminDocs
WHERE id = 1;

-- XQuery 5: Get numbers > 500 using predicate in FOR clause
-- Filter in for clause instead of where clause
SELECT xDoc.query(' 
    for $prod in //product[number > 500] 
    let $x := $prod/number
    return (<Item>{$x}</Item>)
') AS items_gt_500
FROM AdminDocs
WHERE id = 1;

-- XQuery 6: Get numbers > 500 with data() function
-- data() extracts text value without XML tags
SELECT xDoc.query(' 
    for $prod in //product 
    let $x := $prod/number
    where $x > 500
    return (<Item>{data($x)}</Item>)
') AS items_with_data
FROM AdminDocs
WHERE id = 1;

-- XQuery 7: Conditional RETURN with IF-THEN-ELSE
-- Different XML elements based on condition
SELECT xDoc.query(' 
    for $prod in //product 
    let $x := $prod/number
    return if ($x > 500) 
           then <book>{data($x)}</book>
           else <paper>{data($x)}</paper>
') AS conditional_result
FROM AdminDocs
WHERE id = 1;

-- PART 4: XML DML (MODIFY OPERATIONS)

-- View document before modification
SELECT * FROM AdminDocs WHERE id = 2;

-- Example 1: INSERT - Add a new section after section 1
UPDATE AdminDocs 
SET xDoc.modify('
    insert 
    <section num="2"> 
        <title>Background</title>
    </section>
    after (/doc//section[@num=1])[1]
')
WHERE id = 2;

-- View document after insertion
SELECT * FROM AdminDocs WHERE id = 2;

-- Example 2: DELETE - Remove section 2
UPDATE AdminDocs 
SET xDoc.modify('
    delete //section[@num="2"]
')
WHERE id = 2;

-- View document after deletion
SELECT * FROM AdminDocs WHERE id = 2;

-- Additional XQuery Examples

-- Create a summary report
SELECT xDoc.query('
    <ProductSummary>
    {
        for $prod in //product
        return <Product>
                   <Dept>{data($prod/@dept)}</Dept>
                   <Number>{data($prod/number)}</Number>
                   <Name>{data($prod/name)}</Name>
               </Product>
    }
    </ProductSummary>
') AS summary
FROM AdminDocs
WHERE id = 1;

-- Group products by department
SELECT xDoc.query('
    for $dept in distinct-values(//product/@dept)
    return <Department name="{$dept}">
           {
               for $prod in //product[@dept=$dept]
               return <ProductNumber>{data($prod/number)}</ProductNumber>
           }
           </Department>
') AS grouped_by_dept
FROM AdminDocs
WHERE id = 1;

-- Products with sorting
SELECT xDoc.query('
    for $prod in //product
    order by $prod/number
    return <Product>{data($prod/number)} - {data($prod/name)}</Product>
') AS sorted_products
FROM AdminDocs
WHERE id = 1;

-- Aggregation example
SELECT xDoc.query('
    <Summary>
        <TotalProducts>{count(//product)}</TotalProducts>
        <MaxNumber>{max(//product/number)}</MaxNumber>
        <MinNumber>{min(//product/number)}</MinNumber>
    </Summary>
') AS aggregate_summary
FROM AdminDocs
WHERE id = 1;

-- USING VALUE() AND EXIST() METHODS


-- VALUE() method: Extract scalar value from XML
SELECT id,
       xDoc.value('(/catalog/product[1]/number)[1]', 'INT') AS first_product_number,
       xDoc.value('(/catalog/product[1]/name)[1]', 'VARCHAR(50)') AS first_product_name
FROM AdminDocs
WHERE id = 1;

-- EXIST() method: Check if XPath expression returns any nodes
SELECT id
FROM AdminDocs
WHERE xDoc.exist('//product[@dept="WMN"]') = 1;

-- Using EXIST in WHERE clause to filter rows
SELECT id,
       xDoc.query('//product[@dept="WMN"]') AS wmn_products
FROM AdminDocs
WHERE xDoc.exist('//product[@dept="WMN"]') = 1;


-- NODES() METHOD - SHREDDING XML


-- Extract XML nodes as relational rows
SELECT id,
       T.c.value('@dept', 'VARCHAR(10)') AS department,
       T.c.value('(number)[1]', 'INT') AS product_number,
       T.c.value('(name)[1]', 'VARCHAR(50)') AS product_name
FROM AdminDocs
CROSS APPLY xDoc.nodes('/catalog/product') AS T(c)
WHERE id = 1;
